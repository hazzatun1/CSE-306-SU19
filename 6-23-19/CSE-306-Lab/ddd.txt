#include <iostream>
#include<fstream>

using namespace std;

int main()
{
   
    int row, column;
   // take row column from file
   ifstream fileInput;
   fileInput.open("input.txt");
   fileInput >> row >> column;
   
   //cout<< row <<"--" <<column;
  
  // declaring a 2D array 
  int ** array;
  
   array=new int* [row];
   
   for (int i=0;i<row; i++){
       
       array[i]=new int [column];
   }
   
   
   //take grid input from the file 
   for(int i=0; i < row; i++){
       
       for(int j=0; j< column; j++){
           
           fileInput>> array[i][j];
       }
   }
  
  
  // Print the taken array
   for(int i=0; i < row; i++){
       
       for(int j=0; j< column; j++){
           
           cout<< array[i][j]<< " ";
       }
       cout << endl;
   }
 
  
    return 0;
}
